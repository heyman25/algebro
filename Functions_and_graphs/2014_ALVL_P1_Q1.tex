\item {[}ALVL/2014/P1/1{]}

The function $f$ is defined by 
\[
f:x\mapsto\frac{1}{1-x},\,x\in\mathbb{R},\,x\neq1,\,x\neq0.
\]

\begin{enumerate}
\item Show that $f^{2}\left(x\right)=f^{-1}\left(x\right)$. \hfill{}{[}4{]}
\item Find $f^{3}\left(x\right)$ in simplified form. \hfill{}{[}1{]}
\end{enumerate}
