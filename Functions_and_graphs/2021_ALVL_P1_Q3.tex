\item {[}ALVL/2021/P1/3{]}

A curve has equation $x^{\frac{1}{2}}+y^{\frac{1}{2}}=3$, for $x>0$
and $y>0$. 
\begin{enumerate}
\item Show that $\frac{dy}{dx}=-\left(\frac{y}{x}\right)^{\frac{1}{2}}.$
\hfill{}{[}2{]}
\item Find the equation of the normal to the curve at the point where $x=1$.
\hfill{}{[}4{]}
\end{enumerate}
