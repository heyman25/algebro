$n$ are positive integers and $m\neq n$.

\item Find $\int\sin2mx\sin2nx\,dx$. \hfill{}{[}3{]}
\item Find $\int_{0}^{\pi}\left(f\left(x\right)\right)^{2}\,dx$.\hfill{}
{[}5{]}
\end{enumerate}
